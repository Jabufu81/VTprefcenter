            document.querySelectorAll('.screen').forEach(screen => {
                screen.classList.remove('active');
            });
            
            // Show target screen
            document.getElementById('screen' + screenNumber).classList.add('active');
        }

        // Screen 1: Student button
        document.getElementById('studentBtn').addEventListener('click', function() {
            document.getElementById('studentCheckmark').classList.add('show');
            
            setTimeout(function() {
                goToScreen(2);
            }, 500);
        });

        // Screen 2: High School button
        document.getElementById('highSchoolBtn').addEventListener('click', function() {
            document.getElementById('highSchoolCheck').classList.add('show');
            
            setTimeout(() => {
                goToScreen(3);
            }, 500);
        });

        // Screen 3: Specific Subjects button
        document.getElementById('specificSubjectsBtn').addEventListener('click', function() {
            document.getElementById('specificSubjectsCheck').classList.add('show');
            
            setTimeout(() => {
                goToScreen(4);
            }, 500);
        });

        // Screen 4: Add Parent functionality
        function addParentField() {
            const addParentContainer = document.getElementById('addParentContainer');
            const emailFieldContainer = document.getElementById('emailFieldContainer');
            
            addParentContainer.classList.add('hidden');
            emailFieldContainer.classList.remove('hidden');
            
            const emailField = document.getElementById('emailField');
            emailField.focus();
        }

        document.getElementById('addParentBtn').addEventListener('click', addParentField);
        document.getElementById('addParentText').addEventListener('click', addParentField);

        // Handle email field placeholder
        document.getElementById('emailField').addEventListener('focus', function() {
            if (this.value === 'Email') {
                this.value = '';
                this.style.color = '#2d3748';
            }
        });

        document.getElementById('emailField').addEventListener('blur', function() {
            if (this.value === '') {
                this.value = 'Email';
                this.style.color = '#a0aec0';
            }
        });
    </script>
